## 附录 B　关键源码清单(接口签名)

```cpp
// 数据结构(include/rsp/graph.hpp)
struct Transition { int to; double cost; };
struct Action { int action_id; std::string name; std::vector<Transition> trans; };
struct Node { int id; std::vector<Action> actions; };
class RobustGraph { int n; int terminal; std::vector<Node> nodes; void validate() const; };

// Bellman 算子(include/rsp/bellman.hpp)
double action_value(const RobustGraph&, int x, int action_index, const std::vector<double>& J);
int    greedy_action(const RobustGraph&, int x, const std::vector<double>& J);
std::vector<double> bellman_update(const RobustGraph&, const std::vector<double>& J);

// Proper policy(include/rsp/proper_policy.hpp)
ProperPolicyResult find_initial_proper_policy(const RobustGraph&);
PolicyCheckResult  check_policy_proper(const RobustGraph&, const std::vector<int>& policy);
std::vector<double> evaluate_proper_policy_dag(const RobustGraph&, const std::vector<int>& policy);

// 三个核心算法
ValueIterationResult  value_iteration(const RobustGraph&, double eps, int max_iter, bool init_with_inf, bool save_history);
PolicyIterationResult policy_iteration(const RobustGraph&, int max_outer_iter);
DijkstraLikeResult    dijkstra_like(const RobustGraph&);

// Baseline / rollout / exhaustive
BaselineResult        deterministic_dijkstra_baseline(const RobustGraph&, DeterministicMode);
RolloutResult         adversarial_rollout(const RobustGraph&, const std::vector<int>& policy,
                                          const std::vector<double>& J, int start, int max_steps);
ExhaustiveSearchResult exhaustive_search(const RobustGraph&);

// 统一运行(include/rsp/runner.hpp)
AlgorithmRunResult run_algorithm(const RobustGraph&, const std::string& name, const AlgorithmOptions& = {});
```

## 附录 D　复现命令与环境

```bash
# 环境:Linux,g++ 12,CMake,Python 3.x(纯标准库)
git clone https://github.com/Luchutong/RobustShortestPath.git && cd RobustShortestPath && git checkout red
cmake -S . -B build -DCMAKE_BUILD_TYPE=Release && cmake --build build -j
ctest --test-dir build --output-on-failure

# 实验 1(toy):
for a in vi pi dijkstra exhaustive baseline_nominal; do ./build/rsp_main --input data/toy_graph.txt --algorithm $a --output results; done
./build/run_robustness --input data/toy_graph.txt --output results --start 0 --max-steps 20

# 实验 3(中规模效率):
python3 experiments/generate_medium_graphs.py --output data/random_graphs --sizes 20 50 100 200 --cases 20 --actions 3 --successors 2 --seed 42
./build/run_runtime --input-dir data/random_graphs --output results --epsilon 1e-9 --max-iter 100000

# 实验 4(鲁棒性) + 4b(陷阱图):
python3 experiments/generate_medium_graphs.py --output data/exp4 --sizes 50 --cases 20 --actions 3 --successors-values 1 2 3 4 5 --seed 42
./build/run_robustness --input-dir data/exp4 --output results --start 0 --max-steps 1000
python3 experiments/generate_trap_graphs.py --output data/trap --cases 20 --seed 42
./build/run_robustness --input-dir data/trap --output results --start 0 --max-steps 200
python3 experiments/paired_winrate.py results/robustness.csv

# 可视化:
python3 visualization/plot_comparison.py --runtime-summary-csv results/runtime_summary.csv \
  --robustness-summary-csv results/robustness_summary.csv --requested-s 2 --actions 3 --output report/figures/runtime_comparison.svg
```
